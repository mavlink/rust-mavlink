10pub fn last_os_error() -> core::Error {
11    from_raw_os_error(errno())
12}
13
14pub fn from_raw_os_error(errno: i32) -> core::Error {
15    use libc::{EBUSY, EISDIR, ELOOP, ENOTDIR, ENOENT, ENODEV, ENXIO, EACCES, EINVAL, ENAMETOOLONG, EINTR, EWOULDBLOCK};
16
17    let kind = match errno {
18        EBUSY | EISDIR | ELOOP | ENOTDIR | ENOENT | ENODEV | ENXIO | EACCES => core::ErrorKind::NoDevice,
19        EINVAL | ENAMETOOLONG => core::ErrorKind::InvalidInput,
20
21        EINTR       => core::ErrorKind::Io(io::ErrorKind::Interrupted),
22        EWOULDBLOCK => core::ErrorKind::Io(io::ErrorKind::WouldBlock),
23        _           => core::ErrorKind::Io(io::ErrorKind::Other),
24    };
25
26    core::Error::new(kind, error_string(errno))
27}
28
29pub fn from_io_error(io_error: io::Error) -> core::Error {
30    match io_error.raw_os_error() {
31        Some(errno) => from_raw_os_error(errno),
32        None => {
33            let description = io_error.description().to_string();
34
35            core::Error::new(core::ErrorKind::Io(io_error.kind()), description)
36        }
37    }
38}

44pub fn errno() -> i32 {
45    #[cfg(any(target_os = "macos", target_os = "ios", target_os = "freebsd"))]
46    unsafe fn errno_location() -> *const c_int {
47        extern { fn __error() -> *const c_int; }
48        __error()
49    }
50
51    #[cfg(target_os = "bitrig")]
52    fn errno_location() -> *const c_int {
53        extern {
54            fn __errno() -> *const c_int;
55        }
56        unsafe {
57            __errno()
58        }
59    }
60
61    #[cfg(target_os = "dragonfly")]
62    unsafe fn errno_location() -> *const c_int {
63        extern { fn __dfly_error() -> *const c_int; }
64        __dfly_error()
65    }
66
67    #[cfg(target_os = "openbsd")]
68    unsafe fn errno_location() -> *const c_int {
69        extern { fn __errno() -> *const c_int; }
70        __errno()
71    }
72
73    #[cfg(any(target_os = "linux", target_os = "android"))]
74    unsafe fn errno_location() -> *const c_int {
75        extern { fn __errno_location() -> *const c_int; }
76        __errno_location()
77    }
78
79    unsafe {
80        (*errno_location()) as i32
81    }
82}

84pub fn error_string(errno: i32) -> String {
85    #[cfg(target_os = "linux")]
86    extern {
87        #[link_name = "__xpg_strerror_r"]
88        fn strerror_r(errnum: c_int, buf: *mut c_char, buflen: size_t) -> c_int;
89    }
90    #[cfg(not(target_os = "linux"))]
91    extern {
92        fn strerror_r(errnum: c_int, buf: *mut c_char, buflen: size_t) -> c_int;
93    }
94
95    let mut buf = [0 as c_char; TMPBUF_SZ];
96
97    let p = buf.as_mut_ptr();
98    unsafe {
99        if strerror_r(errno as c_int, p, buf.len() as size_t) < 0 {
100            panic!("strerror_r failure");
101        }
102
103        let p = p as *const _;
104        str::from_utf8(CStr::from_ptr(p).to_bytes()).unwrap().to_string()
105    }
106}
