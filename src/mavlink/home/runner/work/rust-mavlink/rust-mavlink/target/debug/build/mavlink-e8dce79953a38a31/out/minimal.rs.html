794        id: u32,
795        payload: &[u8],
796    ) -> Result<Self, ::mavlink_core::error::ParserError> {
797        match id {
798            HEARTBEAT_DATA::ID => HEARTBEAT_DATA::deser(version, payload).map(Self::HEARTBEAT),
799            PROTOCOL_VERSION_DATA::ID => {
800                PROTOCOL_VERSION_DATA::deser(version, payload).map(Self::PROTOCOL_VERSION)
801            }
802            _ => Err(::mavlink_core::error::ParserError::UnknownMessage { id }),
803        }
804    }
