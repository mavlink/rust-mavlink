26pub struct AsyncSerialConnection {
27    port: Mutex<AsyncPeekReader<SerialStream>>,
28    sequence: AtomicU8,
29    protocol_version: MavlinkVersion,
30    recv_any_version: bool,
31    #[cfg(feature = "signing")]
32    signing_data: Option<SigningData>,
33}

35#[async_trait::async_trait]
36impl<M: Message + Sync + Send> AsyncMavConnection<M> for AsyncSerialConnection {
37    async fn recv(&self) -> Result<(MavHeader, M), crate::error::MessageReadError> {
38        let mut port = self.port.lock().await;
39        let version = ReadVersion::from_async_conn_cfg::<_, M>(self);
40        #[cfg(not(feature = "signing"))]
41        let result = read_versioned_msg_async(port.deref_mut(), version).await;
42        #[cfg(feature = "signing")]
43        let result =
44            read_versioned_msg_async_signed(port.deref_mut(), version, self.signing_data.as_ref())
45                .await;
46        result
47    }
48
49    async fn send(
50        &self,
51        header: &MavHeader,
52        data: &M,
53    ) -> Result<usize, crate::error::MessageWriteError> {
54        let mut port = self.port.lock().await;
55
56        let sequence = self.sequence.fetch_add(
57            1,
58            // Safety:
59            //
60            // We are using `Ordering::Relaxed` here because:
61            // - We only need a unique sequence number per message
62            // - `Mutex` on `self.port` already makes sure the rest of the code is synchronized
63            // - No other thread reads or writes `self.sequence` without going through this `Mutex`
64            //
65            // Warning:
66            //
67            // If we later change this code to access `self.sequence` without locking `self.port` with the `Mutex`,
68            // then we should upgrade this ordering to `Ordering::SeqCst`.
69            atomic::Ordering::Relaxed,
70        );
71
72        let header = MavHeader {
73            sequence,
74            system_id: header.system_id,
75            component_id: header.component_id,
76        };
77
78        #[cfg(not(feature = "signing"))]
79        let result =
80            write_versioned_msg_async(port.reader_mut(), self.protocol_version, header, data).await;
81        #[cfg(feature = "signing")]
82        let result = write_versioned_msg_async_signed(
83            port.reader_mut(),
84            self.protocol_version,
85            header,
86            data,
87            self.signing_data.as_ref(),
88        )
89        .await;
90        result
91    }
92
93    fn set_protocol_version(&mut self, version: MavlinkVersion) {
94        self.protocol_version = version;
95    }
96
97    fn protocol_version(&self) -> MavlinkVersion {
98        self.protocol_version
99    }
100
101    fn set_allow_recv_any_version(&mut self, allow: bool) {
102        self.recv_any_version = allow
103    }
104
105    fn allow_recv_any_version(&self) -> bool {
106        self.recv_any_version
107    }
108
109    #[cfg(feature = "signing")]
110    fn setup_signing(&mut self, signing_data: Option<SigningConfig>) {
111        self.signing_data = signing_data.map(SigningData::from_config)
112    }
113}

115#[async_trait]
116impl AsyncConnectable for SerialConnectable {
117    async fn connect_async<M>(&self) -> io::Result<Box<dyn AsyncMavConnection<M> + Sync + Send>>
118    where
119        M: Message + Sync + Send,
120    {
121        let mut port = tokio_serial::new(&self.port_name, self.baud_rate).open_native_async()?;
122        port.set_data_bits(tokio_serial::DataBits::Eight)?;
123        port.set_parity(tokio_serial::Parity::None)?;
124        port.set_stop_bits(tokio_serial::StopBits::One)?;
125        port.set_flow_control(tokio_serial::FlowControl::None)?;
126
127        Ok(Box::new(AsyncSerialConnection {
128            port: Mutex::new(AsyncPeekReader::new(port)),
129            sequence: AtomicU8::new(0),
130            protocol_version: MavlinkVersion::V2,
131            recv_any_version: false,
132            #[cfg(feature = "signing")]
133            signing_data: None,
134        }))
135    }
